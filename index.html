import Stripe from 'stripe'; const stripe = new Stripe(process.env.STRIPE_SECRET_KEY); export default async function handler(req, res) { // CORS headers res.setHeader('Access-Control-Allow-Origin', 'https://lueursdames.github.io'); // ton frontend exact res.setHeader('Access-Control-Allow-Methods', 'POST, OPTIONS'); res.setHeader('Access-Control-Allow-Headers', 'Content-Type'); if (req.method === 'OPTIONS') { res.status(200).end(); return; } if (req.method !== 'POST') { res.setHeader('Allow', 'POST, OPTIONS'); res.status(405).end('Method Not Allowed'); return; } try { const { product, amount } = req.body; const session = await stripe.checkout.sessions.create({ payment_method_types: ['card'], line_items: [ { price_data: { currency: 'eur', product_data: { name: product || 'Bougie mystique', }, unit_amount: amount || 2000, }, quantity: 1, }, ], mode: 'payment', success_url: 'https://lueursdames.github.io/success.html', cancel_url: 'https://lueursdames.github.io/cancel.html', }); res.status(200).json({ url: session.url }); } catch (err) { console.error(err); res.status(500).json({ error: 'Erreur serveur, veuillez r√©essayer.' }); } }
